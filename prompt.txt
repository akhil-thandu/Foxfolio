from app.db.database import SessionLocal
from app.db.models import Holding, Transaction, IntradaySnapshot, PortfolioSnapshot

db = SessionLocal()
try:
    transactions_deleted = db.query(Transaction).delete()
    holdings_deleted = db.query(Holding).delete()
    intraday_deleted = db.query(IntradaySnapshot).delete()
    snapshots_deleted = db.query(PortfolioSnapshot).delete()
    db.commit()
    print('Cleanup complete.')
    print(f'  Transactions deleted: {transactions_deleted}')
    print(f'  Holdings deleted: {holdings_deleted}')
    print(f'  Intraday snapshots deleted: {intraday_deleted}')
    print(f'  Portfolio snapshots deleted: {snapshots_deleted}')
except Exception as e:
    db.rollback()
    print(f'Cleanup failed: {e}')
    raise
finally:
    db.close()
"
